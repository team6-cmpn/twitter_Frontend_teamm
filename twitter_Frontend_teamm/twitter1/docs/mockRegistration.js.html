import axios from 'axios';

/**
 * post
 * adds to SignUp Modal inputs 
 * @param  payload 
 * @returns sign up completion
 */
export const post = async payload => {
  const URL = `http://localhost:8000/signUpUsers`;
  try {
    const response = await axios(URL, {
      method: 'post',
      data: payload,
    });
    return response.data;
  } catch (error) {
    throw error;
  }
};
